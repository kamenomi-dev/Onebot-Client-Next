安装方式

使用 npm 包管理器：

npm i onebot-client-next

简单例子

import { Client } from "onebot-client-next"

const client = new Client(114514, {
    websocket_address: "ws://127.0.0.1:8080",
    access_token: "452AFF0CC359C86C"
})

client.Start().then(() => {
   // 在成功连接后做点什么... 
});

client.on("message.group", (message) => {
    // 处理消息。
});
